                        # 配置

                        # 查看所有配置及所在文件
git config --list --show-origin

# 查看所有配置
git config --list

# 全局配置
# git提交时的用户名和邮箱
git config --global user.name  
git config --global user.email

# 配置当前项目的用户名和邮箱
git config user.name
git config user.email

                        # 帮助

                        # 查看命令的帮助文档
git help <verb>

# 查看命令的简单介绍
git <verb## -h

                        # 获取仓库

                        # 克隆仓库
git clone 仓库地址 [本地仓库名]

# 创建仓库
git init

                        # 记录每次更新到仓库

                        # 查看当前文件状态
git status

# 查看当前文件状态简览
git status -s

# 跟踪新文件 / 暂存已修改的文件
git add <文件名## | <文件路径## | <表达式>

# 对比已暂存和未暂存的区别
git diff 

# 对比已暂存和最后一次提交的区别
git diff --staged

# 提交暂存区的文件
git commit -m '说明'

# 跳过暂存区提交所有已跟踪的文件修改
git commit -a

# 移除文件
git rm 文件

# 从暂存区中移除
git rm --cached 文件

# 移动文件
git mv file_from file_to

                        # 忽略文件 .gitigonre

                        所有空行或者以 # 开头的行都会被 Git 忽略。

可以使用标准的 glob 模式匹配，它会递归地应用在整个工作区中。

匹配模式可以以（/）开头防止递归。

匹配模式可以以（/）结尾指定目录。

要忽略指定模式以外的文件或目录，可以在模式前加上叹号（!）取反。

                        # 提交记录

                        # 查看提交记录
git log

# 查看提交详情
git log -p

# 指定最新的几次
git log -2

# 查看简略信息
git log --stat

# 指定格式
git log --pretty=oneline
git log --pretty=full
git log --pretty=format

# 显示分支与合并图形
git log --graph

                        # 撤销操作

                        # 追加提交
git commit --amend

# 取消暂存
git reset <file>

# 取消修改，恢复到该文件上次提交时的样子
git checkout <file>

                        # 远程仓库

                        # 查看远程仓库
git remote

# 查看远程仓库和地址
git remote -v

# 添加
git remote add <name## <url>

# 抓取远程数据
git fetch <name>

# 拉去远程数据
git pull <name>

# 推送到远程仓库
git push <remote## <branch>

# 查看某个远程仓库
git remote show <remote>

# 重命名远程仓库
git remote rename <old-remote## <new-remote>

# 移除
git remote remove <remote>

                        # 标签

                        # 查看标签
git tag
git tag -l
git tag --list

# 创建标签
git tag -a <tag-name## -m "标签说明"

# 创建轻量标签
git tag <tag-name>

# 查看标签信息
git show <tag-name>

# 后期打标签
#1.查看对应提交历史的哈希值
#2.在对应的提交历史上打标签
git tag -a <tag-name##  <hash>

# 推送标签到远程仓库
git push <remote## <tag-name>

#推送全部标签
git push <remote## --tags

# 删除本地标签
git tag -d <tag-name>

# 删除远程标签
git tag <remote## --delete <tag-name>

# 检出标签
git checkout <tag-name>

                        # 分支

                        # 创建新分支dev
git branch dev

# 切换到新分支dev
git checkout dev

# 新建分支dev并切换
git checkout -b dev

# 删除分支
git branch -d dev

# 强制删除分支
git branch -D dev

# 查看分支
git branch

# 合并分支 
git merge <branch>

# 查看那些分支合并到当前分支
git branch --merged

# 查看那些分支还没合并到当前分支
git branch --no-merged

                        # 分支使用策略

                        • master 稳定分支
                        • develop 开发分支
                        • topic 主题分支
